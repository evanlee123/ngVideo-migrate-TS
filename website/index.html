ngVideo Angular video player component — migrated from AngularJS

Quick Start

Build the Angular app, then open this page to see the demo:

npm install
ng build
# open website/index.html in a browser

Or run the dev server directly:

ng serve
# visit http://localhost:4200

Component Usage

The video player is composed of Angular components:

<ng-video class="video">

  <vi-controls class="controls"></vi-controls>

  <vi-feedback class="feedback"></vi-feedback>

  <vi-playlist class="playlist"
               [videos]="playlistItems"
               [videoNameFn]="getVideoName"
               (selectVideo)="onVideoSelect($event)"
               (close)="playlistOpen = false">
  </vi-playlist>

</ng-video>

Adding Videos

Use VideoService to add sources to the playlist:

import { VideoService } from './services/video.service';

constructor(private videoService: VideoService) {}

ngOnInit() {
  this.videoService.addSource('mp4', 'https://example.com/video.mp4');
  this.videoService.addSource('mp4', 'https://example.com/video2.mp4');
}

Components

<ng-video>        — Container: hosts <video> element, manages playback
<vi-controls>     — Play/Pause buttons
<vi-feedback>     — Status overlay (time, buffering, loading)
  <vi-volume>     — Volume bar with +/- controls
  <vi-timeline>   — Seekable range slider
  <vi-buffer>     — Canvas-drawn buffer visualization
  <vi-messages>   — Event message counter
<vi-playlist>     — Video list with selection
  <vi-meta>       — Per-video duration metadata
